Algorithmic Efficiency

Optimized Python implementation for the projected earnings calculator:


def calculate_compound_growth(principal, rate, years):
    """
    Calculates future value based on fixed rate
    """
    if years < 0:
        raise ValueError("Time cannot be negative")
        
    future_value = principal * (1 + rate) ** years
    return round(future_value, 2)

# Q3 Projection
result = calculate_compound_growth(1000000, 0.05, 1)
print(f"Projected: ${result}")
